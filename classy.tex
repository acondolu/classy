Valid terms:
\begin{itemize}
    \item $\sigma \colon Var \to ? $
    \item $scopes(t) = \{ \sigma(x) \mid x \mbox{ in } t \}$
    \item If $t_1 * t_2$ and $u_1 * u_2$, then $vars(t_1) \cap vars(u_1)$ iff $vars(t_2) \cap vars(u_2)$
    \item If $\ca{t}{u}$, then $scopes(t)$ disjoint $scopes(u)$
\end{itemize}
